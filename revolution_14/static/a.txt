<?php
$format = 'Format 50 x 50';
$reglage = 'col-sm-4';
$nom_repertoire = 'static/images/fdiv';
$nbcol=3;

	echo '<div class="row">
			<div class="col-md-1"></div>
				<div class="col-md-10">
					<h3>'.$format.'</h3>
';	

//ouverture repertoire
$pointeur = opendir($nom_repertoire);
$i = 0;
//stockage noms des fichiers des images trouvées, dans un tableau
	while ($fichier = readdir($pointeur)) {
	if (substr($fichier, -3) == "gif" || substr($fichier, -3) == "jpg" || substr($fichier, -3) == "png"
	|| substr($fichier, -4) == "jpeg" || substr($fichier, -3) == "PNG" || substr($fichier, -3) == "GIF"
	|| substr($fichier, -3) == "JPG")
	{
		$tab_image[$i] = $fichier;
		$i++;
		}
	}
//fermeture répertoire
	closedir($pointeur);
//on trie le tableau par ordre alphabétique
	array_multisort($tab_image, SORT_ASC);
//affichage des images
   for ($j=0;$j<=$i-1;$j++)
  {
	if($j%$nbcol==0) echo '<div class="row">';
  
      $image = '<a href="'.$nom_repertoire.'/'.$tab_image[$j].'" data-lightbox="roadtrip" data-title="Anne Lavergne, '.$format.'<br />Clic sur le X pour fermer" data-toggle="tooltip" data-placement="top" title="Cliquer pour agrandir">  
	  <img class="img-responsive" src="'.$nom_repertoire.'/'.$tab_image[$j].'" alt="" /></a>';
		echo '<div class="'.$reglage.'">';
		echo ''.$image.'';
		if(autorisation(-127)) {
			echo '<p class="text-center">'.$tab_image[$j].'</p></div>';
		}
		else {	
		echo '<br /></div>';	
		}
	if($j%$nbcol==($nbcol-1))
	
	echo '</div>';	
		}
		echo '		
			</div>
		<div class="col-md-1"></div>
	</div>'; 

?>
