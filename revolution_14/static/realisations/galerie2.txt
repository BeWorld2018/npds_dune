<?php
$format = 'Stands traditionnels';
$reglage = 'col-lg-3 col-md-3';
$nom_repertoire = 'static/images/realisations/galerie2';
$nbcol=4;

	echo '<div class="row">
			<div class="col-md-1"></div>
				<div class="col-md-10">
					<h3>'.$format.'</h3>
';	

//ouverture repertoire
$pointeur = opendir($nom_repertoire);
$i = 0;
//stockage noms des fichiers des images trouvées, dans un tableau
	while ($fichier = readdir($pointeur)) {
	if (substr($fichier, -3) == "gif" || substr($fichier, -3) == "jpg" || substr($fichier, -3) == "png"
	|| substr($fichier, -4) == "jpeg" || substr($fichier, -3) == "PNG" || substr($fichier, -3) == "GIF"
	|| substr($fichier, -3) == "JPG")
	{
		$tab_image[$i] = $fichier;
		$i++;
		}
	}
//fermeture répertoire
	closedir($pointeur);
//on trie le tableau par ordre alphabétique
	array_multisort($tab_image, SORT_ASC);
//affichage des images
   for ($j=0;$j<=$i-1;$j++)
  {
	if($j%$nbcol==0) echo '<div class="row pale">';
$image = '<a href="'.$nom_repertoire.'/'.$tab_image[$j].'" data-lightbox="roadtrip" data-title="&copy; Expo Even<br />'.$format.'" data-toggle="tooltip" data-placement="auto" title="Cliquer pour agrandir"><img class="img-responsive" src="'.$nom_repertoire.'/'.$tab_image[$j].'" alt="" /></a>';
		echo '<div class="'.$reglage.'">';
		echo ''.$image.'';
		if(autorisation(-127)) {
			echo '<p class="text-center">'.$tab_image[$j].'</p></div>';
		}
		else {
		echo '<br /></div>';
		}
	if($j%$nbcol==($nbcol-1))
	
	echo '</div>';
		}
		echo '</div><div class="col-md-1"></div></div>';
?>

  $('[data-toggle="tooltip"]').tooltip()
})
